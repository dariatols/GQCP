117  static UHFSCFEnvironment<Scalar> WithCoreGuess(const size_t N_alpha, const size_t N_beta, const USQHamiltonian<Scalar>& sq_hamiltonian, const ScalarUSQOneElectronOperator<Scalar>& S) {
118 
119  const auto& H_core = sq_hamiltonian.core(); // In AO basis.
120 
121  using MatrixType = Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>;
122  Eigen::GeneralizedSelfAdjointEigenSolver<MatrixType> generalized_eigensolver_a {H_core.alpha().parameters(), S.alpha().parameters()};
123  Eigen::GeneralizedSelfAdjointEigenSolver<MatrixType> generalized_eigensolver_b {H_core.beta().parameters(), S.beta().parameters()};
124  const UTransformationComponent<Scalar> C_initial_a {generalized_eigensolver_a.eigenvectors()};
125  const UTransformationComponent<Scalar> C_initial_b {generalized_eigensolver_b.eigenvectors()};
126  const UTransformation<Scalar> C_initial {C_initial_a, C_initial_b};
127 
128  return UHFSCFEnvironment<Scalar>(N_alpha, N_beta, sq_hamiltonian, S, C_initial);
129  }
